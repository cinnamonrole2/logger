# 1. Dump all CA names
certutil -dump > C:\temp\ca_dump.txt 2>&1

# 2. Extract just the CA names (one per line)
$CAs = certutil -dump | Select-String "CA name" | ForEach-Object { ($_ -split '"')[1] }

# 3. For each CA, dump ALL certificate templates with full details
foreach ($CA in $CAs) {
    Write-Host "Checking CA: $CA" -ForegroundColor Cyan
    certutil -TCAInfo -config "$CA" | Out-File "C:\temp\$($CA -replace '\\','_').txt"
}
